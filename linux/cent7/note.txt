yum修复未完成的事务 
    yum install yum-utils
    yum clean all
    /usr/sbin/yum-complete-transaction --cleanup-only


下载rpm包
    yum -y install yum-utils
    yumdownloader <package> [--destdir .|/path]

查看历史记录
    yum  history list
    yum history info <id>
    yum history redo <id>
    yum history undo <id>  # 撤销

设置时区
    timedatectl set-local-rtc 1#硬件时钟设置与本地时钟一致
    timedatectl list-timezones
    timedatectl set-timezone Asia/Shanghai

设置主机名
    hostnamectl set-hostname mongo1

rpm
    rpm -ivh *.rpm 
    rpm -Uvh *.rpm  # 升级安装

    rpm -ql <package> # 列出所有文件

yum 
    yum deplist <package> 查看包依赖
